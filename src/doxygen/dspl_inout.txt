/*!
	\ingroup IN_OUT_GROUP
	\fn	int dspl_writebin(double* x, double *y, int n,  char* fn)
	\brief Save real or complex вектор to a bin-file `fn`
 
	This function writes real or complex vector `[n x 1]` to a bin-file `fn`. <BR>
	File structure supports 1D and 2D arrays (vectors and matrices). File structure:
	Файл является универсальным для хранения как одномерных, так и двумерных массивов и имеет следующий формат: <BR>
	 type   4 bytes type int. This parameter value can be `DSPL_DAT_COMPLEX`, if pointer `y` is not `NULL`, or `DSPL_DAT_REAL`, if pointer `y` is `NULL`.<BR>
	 `n`		4 bytes type int. Input array rows number. <BR>
	 `m`		4 bytes type int. Input array rows number. This parameter is 1 for 1D input array.<BR>
	 `pR`		`n*m*sizeof(double)` Real part data.<BR>
	 `pI`		`n*m*sizeof(double)` Image part data if `y` is not `NULL`.<BR>
	
	Writed file can be used for algorithm verification by an external program like GNU Octave or Matlab.
	\param[in] x	First array pointer. Vector size is `[n x 1]`.
 
	\param[in] y	Second array pointer. Vector size is `[n x 1]`. Can be `NULL`. 
					Text file will contain only x vector if `y == NULL`

	\param[in] n	Input vector size

	\param[in] fn 	File name
 
	\return
	DSPL_OK if file has been saved successfully.
	DSPL_ERROR_PTR if `x==NULL`.
	DSPL_ERROR_SIZE if `n<1`.
	DSPL_ERROR_FNAME if `fn==NULL`.
	DSPL_ERROR_FOPEN if file `fn` cannot be created.
	
	
	\author
	Sergey Bakhurin.
 	www.dsplib.org	
	

*/









/*!
	\ingroup IN_OUT_GROUP
	\fn	int dspl_writetxt(double* x, double *y, int n, char* fn)
	\brief Save data to text file.
 
	This function saves a real or complex vector to the txt file `fn`. <BR>
	Result file has format<BR>
	x[0]	y[0]<BR>
	x[1]	y[1]<BR>
	...<BR>
	x[n-1]	y[n-1]<BR>
	
	This file can be used for plotting data by external programs.


	
	\return
	DSPL_OK if file has been saved successfully.
	DSPL_ERROR_PTR if `x==NULL`.
	DSPL_ERROR_SIZE if `n<1`.
	DSPL_ERROR_FNAME if `fn==NULL`.
	DSPL_ERROR_FOPEN if file `fn` cannot be created.
	
	\note 
	This function rounds data for saving. 
	So this function cannot be used for accuracy verification

	
	\author
	Sergey Bakhurin. 	www.dsplib.org	
	
	
	
*/

/*!
	\ingroup IN_OUT_GROUP
	\fn	int dspl_writebin(double* x, double *y, int n,  char* fn)
	\brief Сохранить вектор в бинарный файл
 
	Функция сохраняет реальный или комплексный вектор размера `[n x 1]` в бинарный файл `fn`. <BR>
	Файл является универсальным для хранения как одномерных, так и двумерных массивов и имеет следующий формат: <BR>
	 type   4 байта типа int. Может принимать значение `DSPL_DAT_COMPLEX`, если указатель `y` не равен `NULL`; `DSPL_DAT_REAL`, если указатель `y` равен `NULL`.<BR>
	 `n`		4 байта типа int. Количество строк данных. <BR>
	 `m`		4 байта типа int. Количество столбцов данных. При сохранении вектора всегда равно 1<BR>
	 `pR`		`n*m*sizeof(double)` Массив реальной части.<BR>
	 `pI`		`n*m*sizeof(double)` Массив мнимой части. Только если `y` не равен `NULL`.<BR>
	
	Файл может быть использован для верификации алгоритмов сторонними пакетами такими как GNU Octave или Matlab.<BR>
 
	\param[in] x	Указатель на первый вектор. Размер вектора `[n x 1]`.
 
	\param[in] y	Указатель на второй вектор. Размер вектора  `[n x 1]`. Может быть `NULL`. 
					Файл будет содержать только один столбец соответствующий вектору `x` если `y == NULL`

	\param[in] n	Размер входных векторов

	\param[in] fn 	Имя файла
 
	\return
	DSPL_OK Файл сохранен успешно.<BR>
	DSPL_ERROR_PTR если `x==NULL`.<BR>
	DSPL_ERROR_SIZE если `n<1`.<BR>
	DSPL_ERROR_FNAME если `fn==NULL`.<BR>
	DSPL_ERROR_FOPEN если файл `fn` не может быть создан.<BR>
	
	\note 
	Данная функция производит запись в файл без потери точности, поэтому рекомендуется использовать ее для верификации данных DSPL.
	
	\author
	Бахурин Сергей.
 	www.dsplib.org	
	

*/
